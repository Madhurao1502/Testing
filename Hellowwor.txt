removeWave(wave: any) {
  // Find the original index in the left grid data where the wave should be inserted back
  const originalIndex = this.date.findIndex(d => d.StartWeek === wave.Start_Week);

  // Construct the wave object to add back
  const waveToAdd = {
    MMYY: this.date[originalIndex]?.monthYear, // Use optional chaining in case find returns undefined
    Start_Week: wave.Start_Week,
    Start_Date: this.date[originalIndex]?.StartDate,
    End_Date: this.date.find(d => d.StartWeek === wave.End_Week)?.endDate
  };

  // Add wave back to the exact index in the left grid
  if (originalIndex > -1) {
    this.leftGridData.splice(originalIndex, 0, waveToAdd); // Insert the waveToAdd back into the correct index
  } else {
    // Fallback if original index is not found, just push to the end
    this.leftGridData.push(waveToAdd);
  }

  // Remove wave from right grid
  this.rightGridData = this.rightGridData.filter(w => w !== wave);

  // Clear selection
  this.selection.clear();
}
