/**
 * Main expand/collapse all toggle
 */
async expandAllRows(): Promise<void> {
  if (this.isExpandingAll) return;

  this.isExpandingAll = true;
  this.commonService.show('expand-all-rows');

  try {
    if (this.allExpanded) {
      // Collapse all
      this.collapseAllRows();
      this.allExpanded = false;
    } else {
      // Expand all
      await this.expandAllRowsSequentially();
      this.allExpanded = true;
    }
  } catch (error) {
    console.error('Error in expandAllRows:', error);
  } finally {
    this.isExpandingAll = false;
    this.commonService.hide('expand-all-rows');
    this.cdr.detectChanges();
  }
}

/**
 * Collapse all expanded rows and remove child rows from datasource
 */
private collapseAllRows(): void {
  this.commonService.show('collapse-all-rows');
  
  try {
    // Get all detail row directives
    const detailRowsArray = this.detailRows?.toArray() || [];
    
    // Collapse each expanded directive
    detailRowsArray.forEach(detailRow => {
      if (detailRow.isExpanded) {
        detailRow.collapse();
      }
    });

    // Remove all child rows from datasource
    const filteredData = this.dataSource.data.filter(row => !row.isChildRow);
    this.dataSource.data = filteredData;

    // Clear tracking sets
    this.expandedRows.clear();
    this.selectedRows.clear();

    // Trigger change detection to update UI (carets, etc.)
    this.cdr.detectChanges();
  } catch (error) {
    console.error('Error collapsing all rows:', error);
  } finally {
    this.commonService.hide('collapse-all-rows');
  }
}

/**
 * Expand all rows on current page sequentially
 */
private async expandAllRowsSequentially(): Promise<void> {
  // First, collapse all to ensure clean state
  this.collapseAllRows();

  const currentPageData = this.getCurrentPageData();
  const detailRowsArray = this.detailRows?.toArray() || [];

  if (detailRowsArray.length === 0 || currentPageData.length === 0) {
    console.warn('No rows to expand');
    return;
  }

  this.commonService.show('load-all-subgrid-data');

  try {
    // Process each row sequentially
    for (let i = 0; i < Math.min(currentPageData.length, detailRowsArray.length); i++) {
      const parentRow = currentPageData[i];
      const detailRowDirective = detailRowsArray[i];

      // Skip if row is a child row
      if (parentRow.isChildRow) continue;

      // Skip if directive is already expanded
      if (detailRowDirective.isExpanded) continue;

      try {
        // Load child data for this row
        const responseData = await this.loadSubgridDataForRow(parentRow);

        // Insert child rows into datasource
        this.insertChildRowsIntoDataSource(parentRow, responseData.data || [], responseData.weekType || 'current');

        // Mark the directive as expanded
        detailRowDirective.isExpanded = true;
        this.expandedRows.add(detailRowDirective);
        this.selectedRows.add(parentRow);

        // Small delay to prevent overwhelming the UI
        await this.delay(50);
      } catch (error) {
        console.error(`Error expanding row ${i}:`, error);
        // Continue with next row even if this one fails
      }
    }

    // Final change detection
    this.cdr.detectChanges();
  } catch (error) {
    console.error('Error in expandAllRowsSequentially:', error);
  } finally {
    this.commonService.hide('load-all-subgrid-data');
  }
}

/**
 * Insert child rows directly into the table's datasource at the correct position
 */
private insertChildRowsIntoDataSource(parentRow: any, childData: any[], weekType: string): void {
  const currentData = this.dataSource.data;
  
  // Find parent row index (must not be a child row itself)
  const parentIndex = currentData.findIndex(row => 
    row.StoreNumber === parentRow.StoreNumber && !row.isChildRow
  );
  
  if (parentIndex === -1) {
    console.warn('Parent row not found in datasource');
    return;
  }
  
  // Remove any existing child rows for this parent first
  this.removeChildRowsFromDataSource(parentRow);
  
  // If no child data, just return
  if (!childData || childData.length === 0) {
    return;
  }
  
  // Map child data to match parent column structure
  const childRows = childData.map((child, index) => ({
    ...child,
    isChildRow: true,
    isFirstChild: index === 0,
    isLastChild: index === childData.length - 1,
    parentStoreNumber: parentRow.StoreNumber,
    
    // Map child fields to parent column names for seamless alignment
    StoreNumber: '', // Empty for indent
    FldTerrtyNbr: child.bfd || '',
    PositionName: child.taskName || '',
    Bfd: child.bfd || '',
    OutletType: '',
    StoreName: '',
    AddrLine1: '',
    City: '',
    State: '',
    Zip: '',
    AssignedTo: child.assignedTo || '',
    assignedTo: child.assignedTo || '',
    
    // Comments - always empty for child rows
    UserComment: '',
    
    // Task-level data
    TaskCompleted: child.taskCompleted || child.TaskCompleted,
    taskCompleted: child.taskCompleted || child.TaskCompleted,
    GuaranteedMiles: child.guaranteedMiles,
    guaranteedMiles: child.guaranteedMiles,
    CostOver: child.costOver,
    costOver: child.costOver,
    Quality: child.quality,
    
    // Week-specific fields
    ...(weekType === 'future' && {
      AvgCost: child.avgCost,
      avgCost: child.avgCost,
      Wave: child.wave,
      ExpectedCollectionTime: child.expectedCollectionTime
    }),
    
    ...(weekType === 'previous' && {
      WK1FSRProdCost: child.wK1FSRProdCost,
      wK1FSRProdCost: child.wK1FSRProdCost,
      WK1StoreTravelCost: child.wK1StoreTravelCost,
      wK1StoreTravelCost: child.wK1StoreTravelCost,
      WK1StoreTravelOvgCost: child.wK1StoreTravelOvgCost,
      wK1StoreTravelOvgCost: child.wK1StoreTravelOvgCost,
      WK2FSRProdCost: child.wK2FSRProdCost,
      wK2FSRProdCost: child.wK2FSRProdCost,
      WK2StoreTravelCost: child.wK2StoreTravelCost,
      wK2StoreTravelCost: child.wK2StoreTravelCost,
      WK2StoreTravelOvgCost: child.wK2StoreTravelOvgCost,
      wK2StoreTravelOvgCost: child.wK2StoreTravelOvgCost
    }),
    
    // Store task-specific data for dialog access
    storeNumber: child.storeNumber || parentRow.StoreNumber,
    storeName: child.storeName || parentRow.StoreName,
    taskNumber: child.taskNumber,
    taskName: child.taskName
  }));
  
  // Get updated data array and insert child rows after parent
  const updatedData = [...currentData];
  updatedData.splice(parentIndex + 1, 0, ...childRows);
  
  // Update datasource (this triggers table re-render)
  this.dataSource.data = updatedData;
}

/**
 * Remove all child rows for a specific parent from datasource
 */
private removeChildRowsFromDataSource(parentRow: any): void {
  const filteredData = this.dataSource.data.filter(row => 
    !(row.isChildRow && row.parentStoreNumber === parentRow.StoreNumber)
  );
  this.dataSource.data = filteredData;
}

/**
 * Get current page data (only parent rows)
 */
private getCurrentPageData(): any[] {
  if (!this.paginator) {
    return this.dataSource.data.filter(row => !row.isChildRow);
  }
  
  const startIndex = this.paginator.pageIndex * this.paginator.pageSize;
  const endIndex = startIndex + this.paginator.pageSize;
  
  // Filter to only get parent rows on current page
  return this.dataSource.data
    .filter(row => !row.isChildRow)
    .slice(startIndex, endIndex);
}

/**
 * Utility delay method
 */
private delay(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms));
}

/**
 * Check if a row is expanded (for caret animation)
 */
isRowExpanded(element: any): boolean {
  return this.selectedRows.has(element);
}

/**
 * Reset expansion state (called on data refresh, pagination, etc.)
 */
private resetExpansionState(): void {
  this.allExpanded = false;
  this.isExpandingAll = false;
  this.expandedRows.clear();
  this.selectedRows.clear();
  
  // Remove all child rows from datasource
  const parentRowsOnly = this.dataSource.data.filter(row => !row.isChildRow);
  this.dataSource.data = parentRowsOnly;
  
  this.cdr.detectChanges();
}
