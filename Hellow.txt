function extractEndDate(input: string): string | null {
    // Define a regular expression to match the date in the format MM/DD/YYYY
    const datePattern = /(\d{2}\/\d{2}\/\d{4})\s-\s(\d{2}\/\d{2}\/\d{4})/;
    
    // Match the input string against the pattern
    const match = input.match(datePattern);
    
    if (match && match[2]) {
        // Return the end date (second match group)
        return match[2];
    }
    
    // Return null if no match was found
    return null;
}
