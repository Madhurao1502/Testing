setCompleteDayDropdown(completeDay: number) {
    switch (completeDay) {
      case 1:
      case 8:
        this.selectedCompleteDay = '1';
        break;
      case 2:
      case 9:
        this.selectedCompleteDay = '2';
        break;
      case 3:
      case 10:
        this.selectedCompleteDay = '3';
        break;
      case 4:
      case 11:
        this.selectedCompleteDay = '4';
        break;
      case 5:
      case 12:
        this.selectedCompleteDay = '5';
        break;
      case 6:
      case 13:
        this.selectedCompleteDay = '6';
        break;
      case 7:
      case 14:
        this.selectedCompleteDay = '7';
        break;
      default:
        this.selectedCompleteDay = '-1';
    }
  }
