validateFile(file: File) {
  return new Promise((resolve, reject) => {
    if (!file) {
      reject("No file selected.");
      return;
    }

    const reader = new FileReader();
    reader.onload = (event) => {
      const result = event.target?.result as string;
      if (result) {
        const firstLine = result.split("\n")[0]; // Get the first line
        const columns = firstLine.split("|"); // Split by "|"

        if (columns.length < 2) {
          reject("Invalid file format. Not enough columns.");
        } else {
          resolve(`Success | ${columns.length}`);
        }
      } else {
        reject("Error reading file.");
      }
    };

    reader.onerror = () => reject("Error loading file.");
    reader.readAsText(file);
  });
}
