calculateMinDueDate(endDate: string): string {
  const date = new Date(endDate);
  date.setDate(date.getDate() - 7);
  return this.formatToDateTimeString(date);
}

calculateMaxDueDate(endDate: string): string {
  const date = new Date(endDate);
  date.setDate(date.getDate() + 7);
  return this.formatToDateTimeString(date);
}

formatToDateTimeString(date: Date): string {
  const isoString = date.toISOString();
  return isoString.slice(0, 16);  // Extract 'YYYY-MM-DDTHH:MM' portion
}
