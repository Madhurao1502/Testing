implements OnInit {
  taskSpecForm: FormGroup;
  visibleSection: string = '';

  constructor(private fb: FormBuilder) {}

  ngOnInit() {
    this.taskSpecForm = this.fb.group({
      collectOnDisplay: [false],
      collectOnBFD: [false],
      notSampleStore: [false],
      collectOnDay: [''],
      projectOverview: [''],
      locationsToCollect: [''],
      collectionSteps: [''],
      alerts: [''],
      definitions: [''],
      changes: [''],
      miscName: [''],
      misc: [''],
      collCatDetails: ['']
    });
  }

  toggleSection(section: string) {
    if (this.visibleSection === section) {
      this.visibleSection = '';
    } else {
      this.visibleSection = section;
    }
  }

  onSubmit() {
    console.log(this.taskSpecForm.value);
    // Handle form submission
  }
}
